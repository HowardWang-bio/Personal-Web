    <div class="container">
        <h2>Glucose Unit Converter</h2>

        <!-- mg/dL to mmol/L Conversion -->
        <div class="form-group spacing">
            <label for="mgdl">Enter glucose value in mg/dL:</label>
            <input type="number" id="mgdl" placeholder="mg/dL">
            <button onclick="convertToMmol()">Convert to mmol/L</button>
        </div>

        <!-- mmol/L to mg/dL Conversion -->
        <div class="form-group">
            <label for="mmol">Enter glucose value in mmol/L:</label>
            <input type="number" id="mmol" placeholder="mmol/L">
            <button onclick="convertToMgdl()">Convert to mg/dL</button>
        </div>

        <p id="result"></p>
    </div>

    <script>
        // Function to convert mg/dL to mmol/L
        function convertToMmol() {
            var mgdl = document.getElementById("mgdl").value;
            if (mgdl) {
                var mmol = (mgdl / 18).toFixed(2); // Conversion factor
                document.getElementById("result").innerText = mgdl + " mg/dL is equal to " + mmol + " mmol/L";
            } else {
                document.getElementById("result").innerText = "Please enter a valid mg/dL value.";
            }
        }

        // Function to convert mmol/L to mg/dL
        function convertToMgdl() {
            var mmol = document.getElementById("mmol").value;
            if (mmol) {
                var mgdl = (mmol * 18).toFixed(2); // Conversion factor
                document.getElementById("result").innerText = mmol + " mmol/L is equal to " + mgdl + " mg/dL";
            } else {
                document.getElementById("result").innerText = "Please enter a valid mmol/L value.";
            }
        }
    </script>
