            <input type="number" id="known" placeholder="Known glucose (B) in mg/dL">

            <label for="meter">Enter glucose meter reading (C) in mg/dL:</label>
            <input type="number" id="meter" placeholder="Glucose meter reading (C) in mg/dL">

            <button onclick="calculateGlucoseMixture()">Calculate Unknown Glucose (A)</button>
        </div>

        <p id="result"></p>

    <script>
        // Function to calculate unknown glucose (A)
        function calculateGlucoseMixture() {
            var known = document.getElementById("known").value;
            var meter = document.getElementById("meter").value;
            if (known && meter) {
                var unknown = (2 * meter - known).toFixed(2); // Formula to calculate A
                document.getElementById("result").innerText = "The unknown glucose value (A) is " + unknown + " mg/dL.";
            } else {
                document.getElementById("result").innerText = "Please enter valid values for known glucose (B) and meter reading (C).";
            }
        }
    </script>
