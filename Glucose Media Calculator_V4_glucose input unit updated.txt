    <script>
        function toggleGlucoseInput() {
            var glucoseUnit = document.getElementById("glucoseUnit").value;
            var currentGlucoseInput = document.getElementById("currentGlucose");

            if (glucoseUnit === "mg/dL") {
                currentGlucoseInput.placeholder = "mg/dL";
            } else if (glucoseUnit === "mM") {
                currentGlucoseInput.placeholder = "mM";
            }
        }

        function formatVolume(volume) {
            if (volume >= 1000) {
                return (volume / 1000).toFixed(2) + ' L';
            } else if (volume >= 1) {
                return volume.toFixed(2) + ' mL';
            } else {
                return (volume * 1000).toFixed(2) + ' µL';
            }
        }

        function calculateVolume() {
            // Get user inputs
            var glucoseUnit = document.getElementById("glucoseUnit").value;
            var currentGlucose = parseFloat(document.getElementById("currentGlucose").value);
            var finalGlucose = parseFloat(document.getElementById("finalGlucose").value);
            var currentVolume = parseFloat(document.getElementById("currentVolume").value) / 1000; // Convert µL to mL
            var finalVolume = parseFloat(document.getElementById("finalVolume").value) / 1000; // Convert µL to mL
            var numWells = parseInt(document.getElementById("numWells").value); // Number of wells

            var currentGlucoseMMol;

            // Convert current glucose to mmol/L if needed
            if (glucoseUnit === "mg/dL") {
                // Convert current glucose from mg/dL to mmol/L (1 mmol/L = 18 mg/dL)
                currentGlucoseMMol = currentGlucose / 18;
            } else {
                currentGlucoseMMol = currentGlucose;
            }

            // Calculate total glucose needed in final volume (mmol)
            var totalGlucoseRequired = finalGlucose * finalVolume;

            // Calculate total glucose currently in the culture (mmol)
            var totalCurrentGlucose = currentGlucoseMMol * currentVolume;

            // Glucose to be added (mmol)
            var glucoseToAdd = totalGlucoseRequired - totalCurrentGlucose;

            // Volume of new media to add (mL)
            var volumeToAdd = finalVolume - currentVolume;

            // Set up the equations:
            // x + y = volumeToAdd
            // (5 * x + 25 * y) = glucoseToAdd
            var y = (glucoseToAdd - 5 * volumeToAdd) / 20; // Volume of 25mM media (mL)
            var x = volumeToAdd - y; // Volume of 5mM media (mL)

            // Convert to µL for output
            var xMicroLiters = x * 1000;
            var yMicroLiters = y * 1000;

            // Calculate total volumes for all wells
            var totalXMicroLiters = xMicroLiters * numWells;
            var totalYMicroLiters = yMicroLiters * numWells;

            // Display results
            if (xMicroLiters >= 0 && yMicroLiters >= 0) {
                document.getElementById("result").innerText = 
                    `To achieve the desired final glucose concentration of ${finalGlucose.toFixed(2)} mmol/L per well:\n` +
                    `For one well: ${formatVolume(x)} of 5mM glucose media and ${formatVolume(y)} of 25mM glucose media need to be added.\n\n` +
                    `For ${numWells} wells: ${formatVolume(totalXMicroLiters / 1000)} of 5mM glucose media and ${formatVolume(totalYMicroLiters / 1000)} of 25mM glucose media need to be added.`;
            } else {
                document.getElementById("result").innerText = 
                    "Error: Please check your inputs. The desired final glucose concentration cannot be achieved with the given parameters.";
            }
        }
    </script>
